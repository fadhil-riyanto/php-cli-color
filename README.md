# php-cli-color
Simple and easy to use the PHP command-line output of color...

## How to use
```php
<?php
require_once 'color.php';

$tes = new cli_color;
echo $tes->black("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->red("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->green("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->yellow("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->blue("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->magenta("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->cyan("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_gray("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->dark_gray("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_red("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_green("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_yellow("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_blue("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_magenta("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->light_cyan("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->white("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_black("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_red("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_green("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_yellow("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_blue("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_magenta("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_cyan("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_light_gray("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_dark_gray("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_light_red("Testing color class, github.com/fadhil-riyanto").PHP_EOL;

echo $tes->bg_light_yellow("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_light_blue("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_light_magenta("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->bg_light_cyan("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->underline("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->reverse("Testing color class, github.com/fadhil-riyanto").PHP_EOL;
echo $tes->concealed("Testing color class, github.com/fadhil-riyanto").PHP_EOL;

echo $tes->reset().PHP_EOL;
?>
```
